Sub TransformLeaveDataForPostgres()
    Dim ws As Worksheet, wsTarget As Worksheet
    Dim lastRow As Long, lastCol As Long, totalCol As Long, lastDateCol As Long
    Dim r As Long, c As Long, outRow As Long
    Dim empID As Variant, leaveHrs As Variant, cellColor As Long, empTotal As Variant
    Dim dictColors As Object
    Dim targetSheetName As String
    Dim tCol As Integer
    Dim obd As Variant, rsd As Variant ' Variables for date parsing
    
    ' Application optimizations for speed
    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    Application.Calculation = xlCalculationManual

    ' Loop through all sheets in the workbook
    For Each ws In ThisWorkbook.Worksheets
        
        ' Check if sheet name contains "Leave", ignore "MAIN", and ignore previously generated "_result" sheets
        If UCase(ws.Name) <> "MAIN" And InStr(1, ws.Name, "Leave", vbTextCompare) > 0 And Right(ws.Name, 7) <> "_result" Then
            
            targetSheetName = ws.Name & "_result"
            
            ' Delete the target sheet if it already exists
            On Error Resume Next
            ThisWorkbook.Worksheets(targetSheetName).Delete
            On Error GoTo 0
            
            ' Setup Target Worksheet
            Set wsTarget = ThisWorkbook.Worksheets.Add(After:=ws)
            wsTarget.Name = targetSheetName
            
            ' 1. Write Standard Headers to Target (Postgres Friendly)
            Dim headers As Variant
            headers = Array("ID No", "English Name", "Onboard Date", "Resign Date", "Factory", "Group Code", "Department", "Leave Date", "Leave Hour", "Total Hour")
            wsTarget.Range("A1:J1").Value = headers
            
            ' Pre-format specific columns
            wsTarget.Columns("F:F").NumberFormat = "@" ' Force Group Code to Text to keep leading zeros
            wsTarget.Columns("C:D").NumberFormat = "yyyy-mm-dd" ' Format Onboard/Resign dates
            
            ' 2. Map Colors and Build Dynamic Leave Headers (K to X)
            Set dictColors = CreateObject("Scripting.Dictionary")
            Dim targetLeaveCol As Integer
            targetLeaveCol = 11 ' Column K
            
            For c = 8 To 21 ' Columns H to U in Source (14 leave types)
                cellColor = ws.Cells(1, c).Interior.Color
                
                ' Store color mapping: Color -> Target Column Number
                If Not dictColors.Exists(cellColor) Then
                    dictColors.Add cellColor, targetLeaveCol
                End If
                
                ' Write header and copy color to target
                wsTarget.Cells(1, targetLeaveCol).Value = ws.Cells(1, c).Value
                wsTarget.Cells(1, targetLeaveCol).Interior.Color = cellColor
                targetLeaveCol = targetLeaveCol + 1
            Next c
            
            ' Format Headers
            wsTarget.Range("A1:X1").Font.Bold = True
            
            ' 3. Identify Data Boundaries in Source
            lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
            lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column
            
            ' Find the "Total" Column dynamically
            totalCol = 0
            For c = 22 To lastCol ' Search from Col V onwards
                If InStr(1, ws.Cells(1, c).Value, "Total", vbTextCompare) > 0 Then
                    totalCol = c
                    Exit For
                End If
            Next c
            
            ' Determine where dates end based on where Total is found
            If totalCol > 0 Then
                lastDateCol = totalCol - 1
            Else
                lastDateCol = lastCol ' Fallback if no Total column header is explicitly named
            End If
            
            outRow = 2
            
            ' 4. Loop Through Employees and Dates
            For r = 2 To lastRow
                empID = ws.Cells(r, 1).Value
                
                ' Skip empty rows
                If Trim(empID) <> "" Then
                    
                    ' Grab Total Hours for this employee
                    If totalCol > 0 Then
                        empTotal = ws.Cells(r, totalCol).Value
                    Else
                        empTotal = 0
                    End If
                    
                    ' Loop through date columns (Col V to end of dates)
                    For c = 22 To lastDateCol
                        leaveHrs = ws.Cells(r, c).Value
                        
                        ' ONLY process if a leave value exists (ignores blanks and zeros)
                        If IsNumeric(leaveHrs) And leaveHrs > 0 Then
                            cellColor = ws.Cells(r, c).Interior.Color
                            
                            ' Copy Employee Base Data (A through G)
                            wsTarget.Range("A" & outRow & ":G" & outRow).Value = ws.Range("A" & r & ":G" & r).Value
                            
                            ' --- FIX: Retain Group Code Leading Zeros ---
                            ' Grab exactly what is displayed (e.g., "0012") instead of the underlying value
                            wsTarget.Cells(outRow, 6).Value = ws.Cells(r, 6).Text
                            
                            ' --- FIX: Format Onboard Date ---
                            obd = ws.Cells(r, 3).Value
                            If Len(CStr(obd)) = 8 And IsNumeric(obd) Then
                                ' Convert "YYYYMMDD" to actual Date
                                wsTarget.Cells(outRow, 3).Value = DateSerial(Left(obd, 4), Mid(obd, 5, 2), Right(obd, 2))
                            End If
                            
                            ' --- FIX: Format Resign Date ---
                            rsd = ws.Cells(r, 4).Value
                            If Len(CStr(rsd)) = 8 And IsNumeric(rsd) Then
                                ' Convert "YYYYMMDD" to actual Date
                                wsTarget.Cells(outRow, 4).Value = DateSerial(Left(rsd, 4), Mid(rsd, 5, 2), Right(rsd, 2))
                            End If
                            
                            ' Write Leave Date (Formatting as standard YYYY-MM-DD for SQL)
                            wsTarget.Cells(outRow, 8).Value = ws.Cells(1, c).Value
                            wsTarget.Cells(outRow, 8).NumberFormat = "yyyy-mm-dd"
                            
                            ' Write Leave Hour and Total Hour
                            wsTarget.Cells(outRow, 9).Value = leaveHrs
                            wsTarget.Cells(outRow, 9).Interior.Color = cellColor ' Match source highlight
                            wsTarget.Cells(outRow, 10).Value = empTotal
                            
                            ' Match color to place the hour in the specific 14 leave type columns
                            If dictColors.Exists(cellColor) Then
                                tCol = dictColors(cellColor)
                                wsTarget.Cells(outRow, tCol).Value = leaveHrs
                                wsTarget.Cells(outRow, tCol).Interior.Color = cellColor
                            End If
                            
                            outRow = outRow + 1
                        End If
                    Next c
                End If
            Next r
            
            ' AutoFit Target Columns
            wsTarget.UsedRange.Columns.AutoFit
            Set dictColors = Nothing
            
        End If
    Next ws
    
    ' Restore Application Settings
    Application.Calculation = xlCalculationAutomatic
    Application.DisplayAlerts = True
    Application.ScreenUpdating = True
    
    MsgBox "Data transformed successfully! Sheets are ready for Postgres export.", vbInformation, "FIFI: Process Finished"
End Sub
